#include <bits/stdc++.h>
using namespace std;
int n , k , x , y , sum;
int main(){
    scanf("%d%d" , &n , &k);
    for(int i =0; i< n; ++i){
        scanf("%d%d" , &x , &y);
        sum += y - x +1;
    }
    sum %=k;
    if(sum != 0) sum = k - sum;
    printf("%d\n" ,sum);
    return 0;
}
