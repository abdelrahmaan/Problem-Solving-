/*
if char A > D  ==>  "Anton"
if char A < D  ==>  "Danik"
if char A == D ==> "Friendship"

*/

#include<bits/stdc++.h>
using namespace std;
int n  , A , D;
char c;
int main ()
{
    scanf("%d" , &n);
    for(int i = 0; i<n; ++i){
            cin>>c;
        if(c == 'A') ++A;
        else ++D;
    }
    if(A < D) printf("Danik\n");
    else if(A > D)  printf("Anton\n");
    else  printf("Friendship\n");
            return 0;
}
