#include <bits/stdc++.h>
using namespace std;
int n , k ,counter ,  goodNum , here = -1;
string s , numbers = "0123456789";
int main(){
  scanf("%d%d" , &n , &k);
    for(int i = 0; i< n; ++i){
        cin >> s;
            for(int  i = 0; i <= k; ++i)
            {
                here = s.find(numbers[i]);
                if(here != -1)
                 {
                    counter ++;
                    here = -1;
                 }
            }
              if(counter >= k+1)
                    goodNum++;
                    counter =0;
                }
    printf("%d\n" , goodNum);
  return 0;
  }
