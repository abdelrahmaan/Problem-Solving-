#include <bits/stdc++.h>
using namespace std;
string s;
set<char>st;
int main(){
    getline(cin , s);
    for(int i= 0; i<s.length(); ++i)
      if(s[i] >= 'a' && s[i] <= 'z')
    //if(s[i]>= 97 && s[i] <= 122)
        st.insert(s[i]);
        printf("%d\n" , st.size());
            return 0;
}
