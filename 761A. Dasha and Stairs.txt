const int N = 1e2+9 , OO = 1000000009;

    scanf("%d%d" ,&even , &odd );
    total = even + odd;
    if(even == 0 && odd == 0){
        puts("NO");
        return 0;
    }
    if( total % 2 != 0){
        e1 = total / 2;
        o1 = e1+1;
    }else{
        e1 = total / 2;
        o1 = e1;
    }
    if( (even == e1 && odd == o1) || (even == o1 && odd == e1) ) puts("YES");
    else puts("NO");
        return 0;
}
