#include <bits/stdc++.h>
using namespace std;
string s , ss;
int main(){
    cin >> s ;
    for(int i = 0; i< s.length(); ++i){
        s[i] = tolower(s[i]);
        if(s[i] != 'a' && s[i] != 'e' && s[i] != 'i' && s[i] != 'o' && s[i] != 'u' && s[i] != 'y')
        ss +=s[i];
    }
    for(int i = 0; i< ss.length(); ++i)
        cout  << '.' << ss[i];
    cout << endl;
   return 0;
}
