#include<bits/stdc++.h>
using namespace std;
int  n , sum;
string s[200006];
int main ()
{
    scanf("%d" , &n);
    for(int i = 0; i< n; ++i)
        cin>>s[i];
        for(int i = 0; i< n; ++i){
        if( s[i] == "Tetrahedron")
        sum +=4;
        if( s[i] == "Cube")
        sum +=6;
        if( s[i] == "Octahedron")
        sum +=8;
        if( s[i] == "Dodecahedron")
        sum +=12;
        if( s[i] == "Icosahedron")
        sum +=20;
        }
printf("%d\n" , sum);

	return 0;
}
