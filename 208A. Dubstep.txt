#include <bits/stdc++.h>
using namespace std;
string s , NewS;
int k;
int main(){
    cin >> s;
    for(int i =0; i< s.length(); ++i){
            if(s[i] == 'W' && s[i+1] == 'U' && s[i+2] == 'B'){
                    NewS += ' ';
                    i+=2;
            }else
                NewS += s[i];
    }
    if(NewS[0] == ' ')
             k = 1;
    else
            k = 0;
    for(k; k< NewS.length(); ++k)
            cout << NewS[k];
            cout << endl;
            return 0;
}
