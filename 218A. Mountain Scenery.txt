#include <bits/stdc++.h>
using namespace std;
int n , k ;
int main() {
	scanf("%d%d" , &n , &k);
	int a[2*n+1];
	for(int i = 0; i < 2*n+1; i++) scanf("%d" , &a[i]);
	for(int i = 0; i < 2*n+1; i++){
		if( i%2 == 1 && k && a[i] > a[i-1]+1 && a[i] > a[i+1]+1){
			k--;
			printf("%d " , a[i]-1);
		}
		else
			printf("%d " , a[i]);
	}
	printf("\n");
	return 0;
}
