/*

In this problem we have to find the minimal possible value of x such that k·x mod 10?=?0 or k·x mod 10?=?r

*/

#include<bits/stdc++.h>
using namespace std;
int  k , r  , ans;
int main ()
{
    scanf("%d%d" , &k , &r);
    for(int i = 1; i <= 10; ++i)
        if((k * i) % 10 == r || ((k * i) % 10 == 0)){
            printf("%d\n" , i);
            break;
        }
            return 0;
}
