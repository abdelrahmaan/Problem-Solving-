#include <bits/stdc++.h>
using namespace std;
char inp[4][4];

bool test(int i, int j)
{
    if(i+1 == 4 || j+1 == 4)    // don't access outside array
        return false;

    //make sure the 4 cells are equal
    return inp[i][j] == inp[i][j+1] && inp[i][j] == inp[i+1][j] && inp[i][j] == inp[i+1][j+1];
}

bool test()
{
    for(int i = 0; i < 4; ++i)
    {
        for(int j = 0; j < 4; ++j)
            if(test(i, j))
                return true;
    }
    return false;
}int main(){
   for(int i = 0; i < 4; ++i)
    {
        for(int j = 0; j < 4; ++j)
            cin>>inp[i][j];
    }


    bool weDidIt = test();

    for(int i = 0; i < 4; ++i)
    {
        for(int j = 0; j < 4; ++j)
        {
            char original = inp[i][j];

            // Do Flip
            if(inp[i][j] == '.')
                inp[i][j] = '#';
            else
                inp[i][j] = '.';

            if(test())
                weDidIt = true;

            inp[i][j] = original;
        }
    }

    if(weDidIt)
        cout<<"YES";
    else
        cout<<"NO";

    return 0;
}
