{
    scanf("%d" , &n);
    for(int i = 0; i < n; ++i){
        scanf("%d%d%d" , &x ,&y , &z);
        if((x == 1 && y == 1 ) || (x == 1 &&z == 1) || (y == 1 && z == 1))
