#include <bits/stdc++.h>
using namespace std;
string s[1005];
int n,flag;
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin >> s[i];
    for(int i = 0; i < n; i++){
        if(s[i][0] == 'O' && s[i][1] == 'O'){
            s[i][0]='+';
            s[i][1]='+';
            cout<<"YES"<<endl;
            flag=1;
            break;
        }
        else if(s[i][3] == 'O' && s[i][4] == 'O'){
            s[i][3]='+';
            s[i][4]='+';
            cout<<"YES"<<endl;
            flag=1;
            break;
        }
    }
    if(!flag){
        cout<<"NO";
        return 0;
    }
    for(int i=0;i<n;i++){
        cout<<s[i]<<endl;
    }
}
