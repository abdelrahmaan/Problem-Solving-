#include <bits/stdc++.h>
using namespace std;
int n , c1 = 1 , c2;
string s[106] ,sC1 , sC2;
int main(){
    scanf("%d" , &n);
    for(int i = 0; i< n; ++i){
    cin >> s[i];
    sC1 = s[0];
    if(s[i] != sC1)
        sC2 = s[i];
        if(s[i] == sC1)
            c1++;
        else
            c2++;
    }
            if(c1 > c2 ) cout << sC1 <<endl;
            else   cout << sC2 <<endl;
   return 0;
}
