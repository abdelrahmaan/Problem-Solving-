/*
we can change all cases into lower or upper 
and compare it's

*/
#include<bits/stdc++.h>
using namespace std;
string s1 , s2;
int main ()
{
    cin >> s1 >> s2;
    for(int i = 0; i<s1.length(); ++i){
        s1[i] = tolower(s1[i]);
        s2[i] = tolower(s2[i]);
    }
        if(s1 < s2 ) printf("-1\n");
            else if(s1 > s2 ) printf("1\n");
                else printf("0\n");
            return 0;
}
