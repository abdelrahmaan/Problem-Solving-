#include <bits/stdc++.h>
using namespace std;
long long n , i , po , bac = 0;
int main(){
    scanf("%lld" ,&n);
    while(n){
        po = pow(2 , i);
        if(po > n)
        {
            n -= po / 2;
            i = 0;
            bac++;
        }
         i++;
    }
    printf("%lld\n" , bac);
        return 0;
}
