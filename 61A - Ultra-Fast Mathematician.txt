#include <bits/stdc++.h>
using namespace std;
string s1 , s2;
int main(){
        cin >> s1 >> s2;
        for(int i =0; i< s1.size(); ++i)
            if(s1[i]== '1' && s2[i] == '0' || s1[i]== '0' && s2[i] == '1')
            cout<< '1';
            else
            cout<< '0';
return 0;
}
