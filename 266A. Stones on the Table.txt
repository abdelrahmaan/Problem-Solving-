/*
minmum num of take cards on the table to be any two cards neighboring stones had different colors

*/

#include<bits/stdc++.h>
using namespace std;
int counter , n;
string s;
int main ()
{
    scanf("%d" , &n);
    cin>> s;
    for(int i = 0; i < n; ++i)
        if(s[i] == s[i+1])
        counter++;
            printf("%d\n" , counter);
            return 0;
}
