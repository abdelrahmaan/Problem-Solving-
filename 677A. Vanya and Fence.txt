/*
the second line if the number <= h ...count +=1

else count +=2
*/


#include<bits/stdc++.h>
using namespace std;
int n , h , number , counter;
int main ()
{
    scanf("%d%d" , &n , &h);
    for(int i =0; i < n; ++i){
        scanf("%d" , &number);
        if(number <= h )
            counter +=1;
        else
            counter +=2;
    }
    printf("%d\n" , counter);
            return 0;
}
