#include <bits/stdc++.h>
using namespace std;
int n  , ans, x , A[100006];
int main(){
     int n;
    scanf("%d",&n);
     ans = n;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        A[x] = 1;
        while(A[ans])
        {
            printf("%d ",ans);
            ans--;
        }
        puts("");
    }

    return 0;
}
