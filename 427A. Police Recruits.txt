#include<bits/stdc++.h>
using namespace std;
int A[1000006] , n , ne , po;
int main ()
{
    scanf("%d" , &n);
    for(int i = 0; i < n; ++i)
        scanf("%d" , &A[i]);
            for(int i = 0; i < n; ++i){
                    if(A[i] > 0)
                        po += A[i];
                    else if(A[i] < 0){
                        if(po > 0)
                            po--;
                        else
                            ne++;
                    }
            }
            printf("%d\n" , ne);
            return 0;
}
