/*
count num of zeros and ones and subtract it's
*/
#include <bits/stdc++.h>
using namespace std;
int n , zeros , ones;
string s;
int main(){
    scanf("%d" , &n);
    cin >> s;
    zeros = count(s.begin(), s.end() , '0');
    printf("%d\n" , abs(zeros - (n - zeros)));
   return 0;
}

#include<bits/stdc++.h>
using namespace std;
int main ()
{
    string ss;
    int n =0, ones=0 , zeros =0;
    scanf("%d" , &n);
    cin >> ss;
    for(int i = 0; i<ss.length(); ++i)
         if(ss[i] =='1')
            ones ++;
         else
            zeros++;
         if(ones >= zeros)
        printf("%d\n" , ones-zeros);
        else
          printf("%d\n" , zeros-ones);
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
int n , x , k;
vector<int>v;
int main(){
    scanf("%d" , &n);
    for(int i =0; i < n; ++i){
        scanf("%d" , &x);
        v.push_back(x);
    }
    while(1){
        if(v[k] == 1 && v[k+1] == 0)
            v.erase(k ,k+1);
        else
         break;
        ++k;
    }
    printf("%d\n" , v.size());
   return 0;
}
